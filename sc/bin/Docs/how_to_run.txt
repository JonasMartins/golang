- MANAGE DATABASE

volume
create -v ./postgres-data:/var/lib/postgresql/data --name sc-postgres-volume postgres

docker run --name sc_dev --volumes-from sc-postgres-volume -e POSTGRES_PASSWORD=_development -p 5432:5432 -d postgres  

create database sc_dev;

create user dev with password '_development';

grant connect on database sc_dev to dev;

grant all on all tables in schema public to dev;

if user does not have permission, change the postgres default user password

alter user postgres with password '';

then change it in the env so the user postgres is now responsible to change the database
